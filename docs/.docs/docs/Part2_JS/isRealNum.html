判断是否是数字

/**
*判断是否是数字
*
**/
isRealNum(val){
    // isNaN()函数 把空串 空格 以及NUll 按照0来处理 所以先去除，
　　if(val === "" || val ==null){
        return false;
　　}
    if(!isNaN(val)){　　　　
    　　// 对于空数组和只有一个数值成员的数组或全是数字组成的字符串，
        // isNaN返回false，例如：'123'、[]、[2]、['123'],isNaN返回false,
        // 所以如果不需要val包含这些特殊情况，则这个判断改写为if(!isNaN(val) && typeof val === 'number' )
        return true; 
　　}else{ 
        return false; 
　　} 
}
